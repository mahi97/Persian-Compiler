برنامه عبارت
// این برنامه برای آزمایش پیاده‌سازی بخش توابع و ساختارها طراحی شده است
// و هیچ ارزش دیگر ندارد :)

ساختار فاکتوریل {
    ثابت صحیح شماره = ۵؛
    صحیح مقدار؛
}

صحیح محسابه_فاکتوریل (صحیح عدد) {
    اگر عدد < ۰ آنگاه برگردان -۱؛
    وگرنه اگر عدد == ۱ آنگاه برگردان ۱؛
    وگرنه {
        صحیح قبلی = محاسبه_فاکتوریل (عدد - ۱)؛
        برگردان قبلی * عدد؛
    }
}

ساختار توان {
ثابت اعشاری پایه = ۲.۰؛
    ثابت صحیح توان = ۵؛
    اعشاری مقدار؛
}

اعشاری محاسبه_توان (اعشاری پایه، صحیح توان) {
    صحیح شمارنده ۰؛
    صحیح نتیجه = ۱؛
    وقتی (شمارنده < توان) {
        نتیجه *= پایه؛
        شمارنده++؛
    }
}

صحیح اصلی () {
    فاکتوریل.مقدار = محاسبه_فاکتوریل(فاکتوریل.شماره)؛
    // فاکتوریل.مقدار باید برابر نتیجه شود!
    توان.مقدار = محاسبه_توان(توان.پایه، توان.توان)؛
    // توان.مقدار باید برابر نتیجه شود!
}
